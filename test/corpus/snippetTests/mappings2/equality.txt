  when (false) { [0] = 0 }
  ["one"] = 1
  ["two"] = 2
}

res1 = x == x

res7 = x == (x) { when (false) { [0] = 0 } ["one"] = 2 }

  new Mapping { when (false) { [0] = 0 } ["one"] = 1 } == 
    new Mapping { when (false) { [0] = 0 } ["one"] = 1 default = 1 }
