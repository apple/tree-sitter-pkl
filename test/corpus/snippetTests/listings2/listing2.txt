import "pkl:test"

function id(x) = x

local x = new Listing {
  when (false) { 0 }
  "one"
  "two"
  "three"
}

res1 = (x) {
  when (false) { 0 }
  [1] = "updated two"
}

res1a = (x) {
  when (false) { 0 }
  [id(1)] = id("updated two")
}

res2 = (x) {
  when (false) { 0 }
  [0] = "updated one"
  [1] = "updated two"
  [2] = "updated three"
}

res2a = (x) {
  when (false) { 0 }
  [id(0)] = id("updated one")
  [1] = id("updated two")
  [id(2)] = "updated three"
}

res3 = (x) {
  when (false) { 0 }
  [2] = "updated three"

  [1] = "updated two"
}

res3a = (x) {
  when (false) { 0 }
  [id(2)] = id("updated three")
  [0] = id("updated one")
  [id(1)] = "updated two"
}
