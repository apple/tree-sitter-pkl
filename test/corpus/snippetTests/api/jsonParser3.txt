import "pkl:json"

local parser = new json.Parser {
  converters {
    ["^firstName"] = (_) -> "changed"
    ["^city"] = (_) -> "no match"
    ["age"] = (_) -> 42
    ["address.city"] = (it) -> it.reverse()
    ["state"] = (_) -> "CA"
    ["phoneNumber[*].type"] = (_) -> "mobile"
    ["phoneNumber[*]"] = (it) -> it.type
    ["gender.*"] = (_) -> "female"
  }
}

// from: https://en.wikipedia.org/wiki/JSON#JSON_sample
res1 = parser.parse("""
