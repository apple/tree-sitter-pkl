local list1 = List(1, "two", 3)
local list2 = List(1, "two", 4)

examples {
  ["equality"] {
    list1 == list1
    list1 == list2
    list1 == list1.toSet()
    list1 == list2.toSet()
  }

  ["inequality"] {
    list1 != list1
    list1 != list2
    list1 != list1.toSet()
    list1 != list2.toSet()
  }

  ["addition"] {
    list1 + list2
    List() + List()
    list1 + List()
    List() + list2
    list1 + list2.toSet()
    List() + Set()
    list1 + Set()
    List() + list2.toSet()
  }

  ["subscript"] {
    list1[0]
    list1[1]
    list1[2]
    module.catch(() -> list1[-1])
    module.catch(() -> list1[3])
  }
}
