function zeroArgs(): String(!isEmpty) = "Pigeon"

function oneArg(a: String(!isEmpty)): String(!isEmpty) = a

function twoArgs(a: String(!isEmpty), b: String(!isEmpty)): String(!isEmpty) = a + b

function mixed(a: String(!isEmpty), b) = a + b
