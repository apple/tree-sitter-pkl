    !equals(42.s, 43.s)

    equals(new Dynamic { name = "Pigeon" }, new Dynamic { name = "Pigeon" })
    !equals(new Dynamic { name = "Pigeon" }, new Dynamic { name = "Parrot" })

    !equals("Pigeon", "Parrot")
    equals(42, 42)
    !equals(42, 43)
    equals(1.23, 1.23)
    !equals(1.23, 1.24)
    equals(true, true)
    !equals(true, false)
  }
}
