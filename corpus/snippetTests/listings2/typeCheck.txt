  when (false) { 0 }
  "pigeon"
  42
  "barn owl"
}

hidden x2: Listing<String(length > 3)> = new {
  when (false) { 0 }
  "pigeon"
  "bob"
}

hidden x3: Listing<String>(!isEmpty)

res1 = test.catch(() -> x1.toList())
res2 = test.catch(() -> x2.toList())
res3 = test.catch(() -> x3.toList())
