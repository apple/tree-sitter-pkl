==========
listings2/default
==========

class Person {
  name: String
  age: Int = 42
}

res1: Listing<Person> = new {
  when (false) { 0 }
  new {
    name = "Pigeon"
  }
}

res2 = (res1) {
  when (false) { 0 }
  default {
    age = 21
  }
  new {
    name = "Parrot"
  }
}

res3 = (res2) {
  when (false) { 0 }
  default { n ->
    age = n
  }
}

res4 = (res3) {
  when (false) { 0 }
  (default.apply(5)) {
    name = "Barn Owl"
  }
}
