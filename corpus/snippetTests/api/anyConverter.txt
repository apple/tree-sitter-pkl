==========
api/anyConverter
==========

class User {
  name: String
}

class Dog {
  age: Int
}

class Env {
  type: "test"|"prod"
}

foo = new Listing {
  new User {
    name = "Pigeon"
  }
  new Dog {
    age = 5
  }
  new Env {
    type = "prod"
  }
  "42"
}

output {
  renderer = new PcfRenderer {
    converters {
      [Any] = (o) -> if (o is module) o else "Unconverted class: \(o.getClass())"
      [Dog] = (d) -> "Dog(\(d.age))"
    }
  }
}
