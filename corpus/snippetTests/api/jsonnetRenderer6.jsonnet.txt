==========
api/jsonnetRenderer6.jsonnet
==========

import "pkl:jsonnet"

name = "pigeon"
age = 42
friends = new Listing {
  "barn owl"
  "parrot"
}
hobbies {
  ["surfing"] {
    skill = "low"
  }
  ["firemaking"] {
    skill = "high"
  }
}
address {
  street = "Norton St."
  zip = 12345
}

output {
  renderer = new jsonnet.Renderer {
    converters {
      ["name"] = (it) -> it.reverse()
      ["friends[*]"] = (it) -> it + "x"
      ["hobbies[*]"] = (it) -> it.skill
      ["address.street"] = (it) -> "Other St."
      ["address.*"] = (it) -> "changed"
      ["unmatched"] = (it) -> "unmatched"
    }
  }
}
