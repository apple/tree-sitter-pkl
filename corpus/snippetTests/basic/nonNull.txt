==========
basic/nonNull
==========

amends "../snippetTest.pkl"

local class Person { name: String = "Pigeon" }

facts {
  ["returns its non-null operand"] {
    123!! == 123
    let (x = 123) x!! == 123
    1.23.ms!! == 1.23.ms
    Pair(21, 42)!! == Pair(21, 42)
    new Person {}!! == new Person {}
  }
}

examples {
  ["fails if operand is null"] {
    module.catch(() -> null!!)
    module.catch(() -> let (x = null) x!!)
  }
}
