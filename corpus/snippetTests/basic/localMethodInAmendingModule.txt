  res1 = x

  bar {
    local function f1(x: Int): Int = x + 2
    res5 = f1(42)
    res6 = test.catch(() -> f1("wrong type"))
  }
  c = new C {}
}

  foo {
    bar {
      local function f1(x: Int): Int = x + 2
      res7 = f1(42)
      res8   = test.catch(() -> f1("wrong type"))
    }
  }
}
