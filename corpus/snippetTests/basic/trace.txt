==========
basic/trace
==========

amends "../snippetTest.pkl"

examples {
  ["trace different values"] {
    trace(1)
    trace(true)
    trace("Barn Owl")
    trace("""
      Bird
      Barn Owl
      """)
    trace(List(1, 2, 3))
    trace(Set(1, 2, 3))
    trace(Map(1, "one", 2, "two"))
    trace(new {
      name = "Barn Owl"
      age = 42
    })
    trace(new {
      "Parrot"
      "Barn Owl"
      "Pigeon"
    })
    trace(new {
      ["Parrot"] {
        name = "Parrot"
        age = 42
      }
    })
  }
}
