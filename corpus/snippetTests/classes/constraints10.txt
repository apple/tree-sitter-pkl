const function matches(pattern: String): (String) -> Boolean = (str) -> str.matches(Regex(pattern))

  a: String(length >= 5, matches("\\w+\\.\\w+")) = "one.two"
}

res1 = new X {}

  a = "five.six"
}

  new X {

  }.a
)
